GIT Push
1. pwd
2. cd 'folder_name' i.e. C:/Users/Dell/Downloads/RandomApplication
3. git init
5. git add and git status
6. git commit -m 'Label Here'
7. git remote origin "https://github.com/culid/Fuentes_Push_Pull.git"
9.git push -u origin master

GIT Pull
1. git pull 'remote_name' 'branch_name'
2. Fork
3. cd then ls
4. git checkout -b 'branch_name'
5. git push origin 'branc_name'
